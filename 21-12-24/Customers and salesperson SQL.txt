1. Identify which salesperson has generated the highest total sales.
1)select sname from salespeople as s,customer as c,orders as o where s.snum=c.snum and c.cnum=o.cnum and o.amt=(select max(o.amt) as highest_sales from orders as o);

2. Count how many customers are handled by each salesperson.
2)select s.snum,count(c.cnum) from customer c,salespeople s where c.snum=s.snum group by s.snum;

3. Find out which customers have placed the most orders.
3)select c.cnum,c.cname,count(o.onum) from customer c,orders o where o.cnum=c.cnum group by c.cnum,c.cname order by count(o.onum) desc limit 1;

4. Analyze the average size of orders for each salesperson's customers.
4)select sname,avg(amt) as avg_amount from customer c,salespeople s,orders o where s.snum=c.snum and c.cnum=o.cnum group by s.snum;

5. Identify the customers who have spent the most money on orders.
5)select c.cname,c.cnum,o.amt from customer c,orders o where o.cnum=c.cnum and o.amt=(select max(o.amt) from orders o);

6. Analyze which cities (regions) generate the highest total sales
6)select c.cnum,c.city,o.amt from customer c,orders o where o.cnum=c.cnum and o.amt=(select max(o.amt) from orders o);

7. Find out which salesperson has the highest average customer rating.
7)select s.sname,avg(rating) from salespeople s,customer c where s.snum=c.snum order by s.snum desc limit 1;

8.Calculate the percentage of customers who have placed orders for each salesperson.
8)SELECT 
    s.snum,
    s.sname,
    s.city,
    ROUND(
        ( 
            (SELECT COUNT(DISTINCT cnum) 
             FROM customer 
             WHERE snum = s.snum AND cnum IN (SELECT DISTINCT cnum FROM orders)
            ) 
            * 100.0
        ) / 
        (SELECT COUNT(*) FROM customer WHERE snum = s.snum), 2
    ) AS order_percentage
FROM salespeople s;


9.Calculate the total revenue generated by each customer
9)select s.snum,s.sname,sum(o.amt) from salespeople s,customer c,orders o where s.snum=c.snum and c.cnum=o.cnum group by s.sname;

10.Identify customers who have not placed any orders
10)SELECT 
    s.snum,
    s.sname,
    s.city,
    ROUND(
        ( 
            (SELECT SUM(o.amt) 
             FROM orders o 
             JOIN customer c ON o.cnum = c.cnum
             WHERE c.snum = s.snum
            ) 
            * 100.0
        ) / 
        (SELECT SUM(amt) FROM orders), 2
    ) AS revenue_percentage
FROM salespeople s;
